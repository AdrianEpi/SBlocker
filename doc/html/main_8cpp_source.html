231 void calculateError (int& argc, char* argv[]) {
232  if (argc != 4) {
233  std::cout << std::endl << "Error, the program needs 4 arguments to calculate the success and error percentage:" << std::endl << "\t bin/textClassifier -e classifiedResume expectedResume" << std::endl;
234  exit(1);
235  }
236  std::string expected = argv[3];
237  std::string received = argv[2];
238  std::vector<std::string> expect;
239  std::vector<std::string> receive;
240  std::string tmp = "";
241  std::ifstream file(expected, std::ios::in);
242  if (file.fail()) {
243  std::cout << std::endl << "Error 404," << expected << " file not found." << std::endl;
244  exit(1);
245  }
246  else {
247  while (!file.eof()) {
248  file >> tmp;
249  expect.push_back(tmp);
250  }
251  file.close();
252  }
253  std::ifstream file2(received, std::ios::in);
254  if (file2.fail()) {
255  std::cout << std::endl << "Error 404," << received << " file not found." << std::endl;
256  exit(1);
257  }
258  else {
259  std::getline(file2, tmp);
260  while (!file2.eof()) {
261  file2 >> tmp;
262  receive.push_back(tmp);
263  }
264  file2.close();
265  }
266 
267  int size = 0;
268  if (expect.size() < receive.size()) {
269  size = expect.size();
270  }
271  else {
272  size = receive.size();
273  }
274  int correct = 0;
275  for (int i = 0; i < size; i++) {
276  if (expect[i] == receive[i]) {
277  correct++;
278  }
279  }
280  float percentage = correct;
281  percentage /= size;
282  percentage *= 100;
283  std::cout << std::endl << "Success percentage: " << correct << " / " << size << " = " << percentage << " %.";
284  std::cout << std::endl << "Error percentage: " << (size - correct) << " / " << size << " = " << (100 - percentage) << " %." << std::endl;
285 }
