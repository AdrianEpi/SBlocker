395 int PreProcesser::loadData (std::string& inputFile, std::string dataType) {
396  data_ = "";
397  int dataLines = 0;
398  std::ifstream file(inputFile, std::ios::in);
399  if (file.fail()) {
400  std::cout << std::endl << "Error 404, file not found in loadData function." << std::endl;
401  exit(1);
402  }
403  std::string aux = "";
404  while (!file.eof()) {
405  file >> aux;
406  if (dataType == "" ) {
407  data_ += aux + " ";
408  std::getline(file, aux);
409  data_ += aux;
410  dataLines++;
411  }
412  else {
413  if (aux.length() < dataType.length()) {
414  std::string spacedName = "";
415  while (aux.length() < dataType.length()) {
416  file >> spacedName;
417  aux += " " + spacedName;
418  }
419  }
420  std::string readedType = aux.substr(0, dataType.length());
421  aux = aux.substr(dataType.length(), aux.length());
422  std::getline(file, aux);
423  if (readedType == dataType) {
424  data_ += aux;
425  dataLines++;
426  }
427  }
428 
429  }
430  file.close();
431  return dataLines;
432 }
