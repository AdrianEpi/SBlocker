281 void Vocabulary::generateVocabulary (std::string& inputFile, bool tokenize) {
282  std::ifstream file(inputFile, std::ios::in);
283  if (file.fail()) {
284  std::cout << std::endl << "Error 404, generateVocabulary file not found. (" << inputFile << ")" << std::endl;
285  exit(1);
286  }
287  set_NTokens(0);
288  set_VocabularyCounter(0);
289  std::string word;
290  std::set<Token>::iterator it;
291  while (!file.eof()) {
292  file >> word;
293  if (isdigit(word[0])) {
294  nLines_ = stoi(word);
295  }
296  else {
297  if (!vocabulary_.count(word)) {
298  Token newToken(word);
299  vocabulary_.insert(newToken);
300  }
301  else if (tokenize){
302  it = vocabulary_.find(word);
303  Token newToken = *it;
304  newToken.incrementate();
305  vocabulary_.erase(word);
306  vocabulary_.insert(newToken);
307  }
308  nTokens_++;
309  }
310  }
311  file.close();
312  set_VocabularyCounter(vocabulary_.size());
313 }
