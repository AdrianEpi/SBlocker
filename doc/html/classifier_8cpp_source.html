171 void Classifier::classify (std::vector<std::string> sentence) {
172  std::vector<float> prob;
173  prob.resize(inputFiles_.size());
174  for (unsigned i = 0; i < prob.size(); i++) {
175  prob[i] = 0.0;
176  }
177  std::set<Token>::iterator it;
178  for (unsigned i = 0; i < sentence.size(); i++) {
179  it = learnedData_.find(sentence[i]);
180  for (unsigned j = 0; j < prob.size(); j++) {
181  prob[j] += it -> get_MultiClass(j);
182  }
183  }
184  data_ += ", ";
185  unsigned selection = 0;
186  for (unsigned i = 0; i < prob.size(); i++) {
187  prob[i] += class_[i].get_ClassProbability();
188  if (prob[selection] < prob[i]) {
189  selection = i;
190  }
191  data_ += std::to_string(prob[i]);
192  data_ += ", ";
193  }
194  resume_.push_back(class_[selection].get_Type());
195  data_ += class_[selection].get_Type();
196  data_ += ".\r";
197 }

238 void Classifier::readInputFiles (char* argv[], int& argc) {
239  std::vector<Token> tmpVector;
240  for (int k = 4; k < argc; k++) {
241  std::string fileName = argv[k];
242  inputFiles_.push_back(fileName);
243  std::string type = "";
244  // ONLY FOR THIS PROGRAM
245  if (fileName == "../outputs/learned_SPAM.txt") {
246  type += "SPAM";
247  }
248  else {
249  type += "HAM";
250  }
251  std::ifstream file(fileName, std::ios::in);
252  if (file.fail()) {
253  std::cout << std::endl << "Error 404," << fileName << " file not found." << std::endl;
254  exit(1);
255  }
256  else if (k == 4) {
257  std::string reader = "";
258  file >> reader >> reader >> reader >> reader >> reader >> reader;
259  Vocabulary newVoc;
260  newVoc.set_NLines(std::stoi(reader));
261  newVoc.set_Type(type);
262  class_.push_back(newVoc);
263  std::getline(file, reader);
264  while (!file.eof()) {
265  file >> reader >> reader;
266  Token newToken(reader);
267  file >> reader >> reader >> reader >> reader;
268  newToken.addClassProb(std::stof(reader), type);
269  newToken.set_Type(type);
270  tmpVector.push_back(newToken);
271  }
272  }
273  else {
274  int i = 0;
275  std::string reader = "";
276  file >> reader >> reader >> reader >> reader >> reader >> reader;
277  Vocabulary newVoc;
278  newVoc.set_NLines(std::stoi(reader));
279  newVoc.set_Type(type);
280  class_.push_back(newVoc);
281  std::getline(file, reader);
282  while (!file.eof()) {
283  file >> reader >> reader >> reader >> reader >> reader >> reader;
284  tmpVector[i].addClassProb(std::stof(reader), type);
285  i++;
286  }
287  }
288  file.close();
289  }
290  for (unsigned i = 0; i < tmpVector.size(); i++) {
291  learnedData_.insert(tmpVector[i]);
292  }
293 }
